function MinMax(position, depth, alpha,beta, maximizingPlayer)
{
    if depth ==0 or game over in position
        return static evaluation of position

    if maximizingPlayer
        maxEval=-infinity
        for each pos in valid position
            eval= MinMax(pos,depth-1,alpha,beta,false)
            maxEval=max(alpha,eval)
            if beta <= alpha
                break
        return maxEval
    else
        minEval= +infinity
        for each pos in valid position
            eval=MinMax(pos,depth-1,alpha,beta,true)
            minEval=min(beta,eval)
            if beta<= alpha
                break
        return minEval
}
